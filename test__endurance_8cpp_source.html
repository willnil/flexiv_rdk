60 void highPriorityTask(std::shared_ptr<flexiv::RobotStates> robotStates,
61  std::shared_ptr<flexiv::Robot> robot)
62 {
63  // read robot states
64  robot->getRobotStates(robotStates.get());
65 
66  // TCP movement control
67  //=====================================================================
68  // use target TCP velocity and acceleration = 0
69  std::vector<double> tcpVel(6, 0);
70  std::vector<double> tcpAcc(6, 0);
71 
72  // set initial TCP pose
73  if (!g_isInitPoseSet) {
74  // check vector size before saving
75  if (robotStates->m_tcpPose.size() == k_cartPoseSize) {
76  g_initTcpPose = robotStates->m_tcpPose;
77  g_currentTcpPose = g_initTcpPose;
78  g_isInitPoseSet = true;
79  }
80  }
81  // run control after initial pose is set
82  else {
83  // move along Z direction
84  g_currentTcpPose[2] = g_initTcpPose[2]
85  + k_swingAmp
86  * sin(2 * M_PI * k_swingFreq
87  * g_hpLoopCounter * k_loopPeiord);
88  robot->streamTcpPose(g_currentTcpPose, tcpVel, tcpAcc);
89  }
90 
91  // save data to global buffer, not using mutex to avoid interruption on RT
92  // loop from potential priority inversion
93  g_logData.tcpPose = robotStates->m_tcpPose;
94  g_logData.tcpForce = robotStates->m_extForceInBaseFrame;
95 
96  // increment loop counter
97  g_hpLoopCounter++;
98 }
